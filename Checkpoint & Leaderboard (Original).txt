-- Server script for persistent checkpoints with DataStore and leaderboard
-- Players get +1 point only once per checkpoint
-- Players spawn at last checkpoint on rejoin
-- "Stage Completed" message removed as requested
-- Place inside ServerScriptService

local DataStoreService = game:GetService("DataStoreService")
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local checkpointStore = DataStoreService:GetDataStore("CheckpointDataStore")
local checkpointsFolder = workspace:WaitForChild("Checkpoints")

-- RemoteEvents for client messages
local checkpointMessageEvent = ReplicatedStorage:FindFirstChild("CheckpointMessageEvent")
if not checkpointMessageEvent then
	checkpointMessageEvent = Instance.new("RemoteEvent")
	checkpointMessageEvent.Name = "CheckpointMessageEvent"
	checkpointMessageEvent.Parent = ReplicatedStorage
end

local levelCompletedMessageEvent = ReplicatedStorage:FindFirstChild("LevelCompletedMessageEvent")
if not levelCompletedMessageEvent then
	levelCompletedMessageEvent = Instance.new("RemoteEvent")
	levelCompletedMessageEvent.Name = "LevelCompletedMessageEvent"
	levelCompletedMessageEvent.Parent = ReplicatedStorage
end

local playerCheckpoints = {}
local checkpointDebounces = {}

local function saveCheckpoint(player, checkpointName)
	local key = "player_" .. player.UserId
	local success, err
	for attempt = 1, 3 do
		success, err = pcall(function()
			checkpointStore:UpdateAsync(key, function(oldValue)
				local newIndex = tonumber(checkpointName:match("%d+"))
				local oldIndex = tonumber(tostring(oldValue or ""):match("%d+") or "0")
				if newIndex and newIndex > oldIndex then
					return checkpointName
				else
					return oldValue
				end
			end)
		end)
		if success then
			print("[CheckpointSystem] Saved checkpoint '" .. checkpointName .. "' for player " .. player.Name)
			return true
		else
			warn("[CheckpointSystem] Attempt " .. attempt .. " failed to save checkpoint for " .. player.Name .. ": " .. tostring(err))
			wait(0.5)
		end
	end
	return false
end

local function loadCheckpoint(player)
	local key = "player_" .. player.UserId
	local success, data = pcall(function()
		return checkpointStore:GetAsync(key)
	end)
	if success then
		if data and checkpointsFolder:FindFirstChild(data) then
			print("[CheckpointSystem] Loaded checkpoint '" .. data .. "' for player " .. player.Name)
			return data
		else
			return nil
		end
	else
		warn("[CheckpointSystem] Failed to load checkpoint for " .. player.Name .. ": " .. tostring(data))
		return nil
	end
end

local function teleportToCheckpoint(player)
	local character = player.Character or player.CharacterAdded:Wait()
	local humanoidRootPart = character:WaitForChild("HumanoidRootPart")

	local checkpointName = loadCheckpoint(player)



	if checkpointName then
		local checkpointPart = checkpointsFolder:FindFirstChild(checkpointName)
		if checkpointPart then
			humanoidRootPart.CFrame = checkpointPart.CFrame + Vector3.new(0,3,0)
			print("[CheckpointSystem] Teleported "..player.Name.." to checkpoint "..checkpointName)
			return
		else
			warn("[CheckpointSystem] Checkpoint part '" .. checkpointName .. "' not found for player " .. player.Name)
		end
	end



	local defaultSpawn = workspace:FindFirstChildOfClass("SpawnLocation") or workspace:FindFirstChild("SpawnLocation")
	if defaultSpawn then
		humanoidRootPart.CFrame = defaultSpawn.CFrame + Vector3.new(0,3,0)
		print("[CheckpointSystem] Teleported "..player.Name.." to default spawn")
	else
		warn("[CheckpointSystem] No spawn location found for " .. player.Name)
	end
end

local function setupLeaderboard(player)
	local leaderstats = Instance.new("Folder")
	leaderstats.Name = "leaderstats"
	leaderstats.Parent = player

	local obstaclesCompleted = Instance.new("IntValue")
	obstaclesCompleted.Name = "Obstacles Completed"
	obstaclesCompleted.Value = 0
	obstaclesCompleted.Parent = leaderstats
end

local function completeCheckpoint(player, checkpointName)
	if not player or not player.UserId then return end

	local debounceKey = player.UserId .. "_" .. checkpointName
	if checkpointDebounces[debounceKey] then
		return
	end
	checkpointDebounces[debounceKey] = true



	local userId = player.UserId
	if not playerCheckpoints[userId] then
		playerCheckpoints[userId] = {}
	end



	if playerCheckpoints[userId][checkpointName] then
		checkpointDebounces[debounceKey] = nil
		return
	end



	local checkpointIndex = tonumber(checkpointName:match("%d+"))
	if not checkpointIndex then
		warn("[CheckpointSystem] Invalid checkpoint name format: " .. checkpointName)
		checkpointDebounces[debounceKey] = nil
		return
	end



	local leaderstats = player:FindFirstChild("leaderstats")
	local stat = leaderstats and leaderstats:FindFirstChild("Obstacles Completed")
	if not stat then
		warn("[CheckpointSystem] Leaderstats/Obstacles Completed missing for "..player.Name)
		checkpointDebounces[debounceKey] = nil
		return
	end



	if stat.Value + 1 == checkpointIndex then
		playerCheckpoints[userId][checkpointName] = true
		stat.Value = checkpointIndex



		local saved = saveCheckpoint(player, checkpointName)
		if not saved then
			warn("[CheckpointSystem] Failed to save checkpoint for "..player.Name.." checkpoint: "..checkpointName)
		end



		-- Removed firing "Stage Completed" message as per request
		-- Fire only 'Level Completed' message
		levelCompletedMessageEvent:FireClient(player, "Level Completed")



		print("[CheckpointSystem] Player "..player.Name.." completed checkpoint "..checkpointName)
	end



	wait(1)
	checkpointDebounces[debounceKey] = nil
end

local function setupCheckpointTouch(checkpointPart)
	checkpointPart.Touched:Connect(function(hit)
		local player = Players:GetPlayerFromCharacter(hit.Parent)
		if not player then return end
		completeCheckpoint(player, checkpointPart.Name)
	end)
end

local function setupAllCheckpoints()
	for _, checkpoint in ipairs(checkpointsFolder:GetChildren()) do
		if checkpoint:IsA("BasePart") then
			setupCheckpointTouch(checkpoint)
		end
	end

	checkpointsFolder.ChildAdded:Connect(function(newCheckpoint)
		if newCheckpoint:IsA("BasePart") then
			setupCheckpointTouch(newCheckpoint)
		end
	end)
end

Players.PlayerAdded:Connect(function(player)
	setupLeaderboard(player)
	playerCheckpoints[player.UserId] = {}

	local leaderstats = player:WaitForChild("leaderstats", 10)
	local stat = leaderstats:WaitForChild("Obstacles Completed", 10)
	if not stat then
		warn("[CheckpointSystem] Obstacles Completed stat missing for player " .. player.Name)
		return
	end



	local savedCheckpointName = loadCheckpoint(player)
	if savedCheckpointName then
		local index = tonumber(savedCheckpointName:match("%d+"))
		if index then
			for i = 1, index do
				playerCheckpoints[player.UserId]["Checkpoint" .. i] = true
			end
			stat.Value = index
			print("[CheckpointSystem] Set Obstacles Completed to " .. index .. " for player " .. player.Name)
		else
			stat.Value = 0
		end
	else
		stat.Value = 0
	end



	player.CharacterAdded:Connect(function(character)
		wait(0.1)
		teleportToCheckpoint(player)
	end)
end)

Players.PlayerRemoving:Connect(function(player)
	playerCheckpoints[player.UserId] = nil
end)

setupAllCheckpoints()

print("[CheckpointSystem] Persistent checkpoint system loaded.")
